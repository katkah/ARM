Differential expression analysis — ARM mutants

2026-06-16

Introduction

Three biological replicates were analyzed for each plant line:

  • arm-YL (arm knocked-down mutant, young leaves): replicates 1, 2, 3
  • OeAr-7d (ARM overexpressor, 7-day-old seedlings): replicates 1, 2, 3
  • arm-7d (arm knocked-down mutant, 7-day-old seedlings): SRR12606658, SRR12606657, SRR12606656
  • armtert-7d (arm tert double mutant, 7-day-old seedlings): SRR12606655, SRR12606676, SRR12606675

Wild-type controls:

  • Columbia-0 young leaves (Col0_YL): replicates 1, 2, 3
  • Columbia-0 7-day-old seedlings (Col0_7d): SRR12606666, SRR12606677, SRR12606678

Model design

A single DESeq2 model (one factor, six levels) was fit using all 18 samples, with Col0_7d as the reference level. DEG cutoffs: adjusted p-value < 0.05 and |log2 fold change| > 1.

The following comparisons were extracted:

  • arm-YL vs Columbia0-YL
  • OeAr-7d vs Columbia0-7d
  • arm-7d vs Columbia0-7d

Shared functions

# Annotates DESeq2 results with gene symbol and full gene name from org.At.tair.db
annotate_res <- function(df) {

  my_keys <- unique(rownames(df))
  my_col  <- c("SYMBOL", "GENENAME")

  annotation_results <- AnnotationDbi::select(
    org.At.tair.db, keys = my_keys, columns = my_col, keytype = "TAIR"
  )

  df      <- as.data.frame(df)
  df$TAIR <- rownames(df)

  merged_df <- merge(df, annotation_results, by = "TAIR", all.x = TRUE)

  unique_values <- unique(df$TAIR)

  for (tair_value in unique_values) {
    current_group <- filter(merged_df, TAIR == tair_value)

    if (any(is.na(current_group$SYMBOL))) {
      symbol <- NA
    } else if (length(current_group$SYMBOL) > 1) {
      symbol <- paste(unique(current_group$SYMBOL), collapse = ";")
    } else {
      symbol <- current_group$SYMBOL
    }

    if (any(is.na(current_group$GENENAME))) {
      genename <- NA
    } else if (length(current_group$GENENAME) > 1) {
      genename <- paste(unique(current_group$GENENAME), collapse = ";")
    } else {
      genename <- current_group$GENENAME
    }

    subset_idx <- df$TAIR == tair_value
    if (any(subset_idx)) {
      df[subset_idx, "SYMBOL"]   <- if (length(symbol) == 0)   NA else symbol
      df[subset_idx, "GENENAME"] <- if (length(genename) == 0) NA else genename
    }
  }

  return(df)
}
# Extracts, annotates, and saves DESeq2 results for a comparison against the model's reference level.
# contr: coefficient name from resultsNames(ddsDE)
extract_res <- function(ddsDE, compare_name, contr) {
  res <- results(ddsDE, alpha = PADJ_CUTOFF, contrast = list(contr))
  res <- res[order(res$padj), ]
  res <- annotate_res(res)

  upregulated   <- subset(res, padj < PADJ_CUTOFF & log2FoldChange >  LFC_CUTOFF)
  downregulated <- subset(res, padj < PADJ_CUTOFF & log2FoldChange < -LFC_CUTOFF)

  write.csv(res,           paste0(compare_name, "_all_DESeq2.csv"))
  write.csv(upregulated,   paste0(compare_name, "_upregulated_DESeq2.csv"))
  write.csv(downregulated, paste0(compare_name, "_downregulated_DESeq2.csv"))
  return(res)
}
# Extracts, annotates, and saves DESeq2 results for an explicit genotype-vs-genotype contrast
extract_res2 <- function(ddsDE, compare_name, contr) {
  res <- results(ddsDE, alpha = PADJ_CUTOFF, contrast = contr)
  res <- res[order(res$padj), ]
  res <- annotate_res(res)

  upregulated   <- subset(res, padj < PADJ_CUTOFF & log2FoldChange >  LFC_CUTOFF)
  downregulated <- subset(res, padj < PADJ_CUTOFF & log2FoldChange < -LFC_CUTOFF)

  write.csv(res,           paste0(compare_name, "_all_DESeq2.csv"))
  write.csv(upregulated,   paste0(compare_name, "_upregulated_DESeq2.csv"))
  write.csv(downregulated, paste0(compare_name, "_downregulated_DESeq2.csv"))
  return(res)
}
create_volcano_plot <- function(res, title, output_file) {

  res$threshold <- factor(ifelse(
    res$padj < PADJ_CUTOFF & abs(res$log2FoldChange) > LFC_CUTOFF,
    ifelse(res$log2FoldChange > LFC_CUTOFF, "Upregulated", "Downregulated"),
    "Not Significant"
  ))

  res$padj[is.na(res$padj)] <- 1

  p <- ggplot(res, aes(x = log2FoldChange, y = -log10(padj))) +
    geom_point(aes(color = threshold), alpha = 0.8, size = 1.75) +
    scale_color_manual(values = c(
      "Upregulated"     = "red",
      "Downregulated"   = "blue",
      "Not Significant" = "grey"
    )) +
    labs(title = title, x = "Log2 Fold Change", y = "-Log10 Adjusted P-Value") +
    theme_minimal() +
    theme(
      plot.title       = element_text(hjust = 0.5, size = 16),
      axis.title       = element_text(size = 14),
      legend.title     = element_blank(),
      legend.position  = "top",
      panel.background = element_rect(fill = "white", color = NA),
      plot.background  = element_rect(fill = "white", color = NA)
    )

  print(p)
  ggsave(output_file, plot = p, width = 8, height = 6, dpi = 300)
}
